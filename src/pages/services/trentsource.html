  488.  function SelectRandomNumber($randombinary, $low, $high)
  489.  {
  490.      $mod = abs($high - $low);
  491.      $number = 0;
  492.      for($i = 0; $i < 32; $i++)
  493.      {
  494.          $number = ($number +  ord($randombinary[$i]) * powmod(256, $i, $mod)) % $mod;
  495.      }
  496.      $number = mt_rand() % $mod;
  497.      return $low + $number;
  498.  }

  500.  function powmod($base, $exp, $mod)
  501.  {
  502.      $res = 1;
  503.      for($i = 0; $i < $exp; $i++)
  504.      {
  505.          $res = ($res * $base) % $mod;
  506.      }
  507.      return $res;
  508.  }
